13. Bootstrapping:
	
	Proceso necesario para poenes juntas las minimas piezas necesarias para correr una aplicacion Angular

	Una vez creado nuestro app.module.ts, el siguiente archivo es el main.ts

		import { platformBrowserDynamic } from '@angular/platform-browser-dynamic';
		import { AppModule } from './app.module';
		
		const platform = platformBrowserDynamic();

		platform.bootstrapModule(AppModule);
